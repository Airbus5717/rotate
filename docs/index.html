Syntax Overview

Basic variables

io :: import("std")

// semicolons are optional
x: int = 1
y := 2 // inferred
z :: 3 // z is const 3
d := ptr(x) // type is int*
d = nil // now NULL

arr : []int = [1, 2, 3] // inferred type ([3]int)
io.println(arr.len) // arr.len == 3

If Else

x := true
y := false
if x {
 // do stuff
} else if !y {
    // do stuff
} else {
    // do stuff
}

Struct

Token :: struct {
    x: int,
    y: [3]int,
}


d := Token{1, [1, 2, 3]}
d.x = 2
d.y[1] = 1

Function

    return x + y

Memory managment

similar to C, with some modern features (new/delete and defer) 

str := new([3]char)
defer delete str

x := new([100]int)
tkn := new(Token) // assuming Token is a struct
defer delete tkn
defer delete x

// how to check for failed allocation
if x == nil {
    io.println("Fail alloc")
    os.exit(1)
}
